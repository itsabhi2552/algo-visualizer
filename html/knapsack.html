                for (Cap = 0; Cap <= Capacity; Cap++):
                       if (i == 0 || w == 0):
                               K[i][Cap] = 0;
                       else if (wt[i - 1] <= Cap):
                               K[i][Cap] = max(pt[i - 1]
                               + K[i - 1][Cap - wt[i - 1]],
                               K[i - 1][Cap])
                        else:
                               K[i][Cap] = K[i - 1][Cap]
